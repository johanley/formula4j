011    /**
012     Represent an immutable number, using a natural, compact syntax. 
013     The number may have a decimal portion, or it may not.
014    
015    <P>This class exists for these reasons:
016    <ul>
017     <li>to simplify calculations, and build on top of what's available from the {@link BigDecimal} class
018     <li>to allow your code to read at a higher level
019     <li>to define a more natural, pleasing syntax
020     <li>to help you avoid floating-point types, 
021     which have many <a href='http://www.ibm.com/developerworks/java/library/j-jtp0114/'>pitfalls</a> 
022    </ul>
023    
024    <P>This class is especially suited for representing money. 
025    For the most common operations, Decimal uses {@link BigDecimal} to perform calculations, and is thus not subject to the 
026    rounding errors which are very common when floating-point primitives are used. 
027    
028    <P><tt>Decimal</tt> objects are immutable.  
029    Many operations in this class return new <tt>Decimal</tt> objects. 
030     
031     <P>As usual, operations can be performed on two items having a different number of decimal places. 
032     For example, these  operations are valid (using an informal, <em>ad hoc</em> notation) : 
033     <PRE>10 + 1.23 = 11.23
034    10.00 + 1.23 = 11.23
035    10 - 1.23 = 8.77
036    (10 > 1.23) => true </PRE> 
037     This corresponds to typical user expectations.
038      
039     <P>The {@link #eq(Decimal)} is usually to be preferred over the {@link #equals(Object)} method. 
040     The {@link #equals(Object)} is unusual, in that it's the only method sensitive to the exact 
041     number of decimal places, while {@link #eq(Decimal)} is not. That is,  
042     <PRE>10.equals(10.00) => false
043    10.eq(10.00) => true</PRE>
044       
045     <P>
046     Various methods in this class have unusually terse names, such as 
047     <tt>lt</tt> for 'less than',  and <tt>gt</tt> for 'greater than', and so on. 
048     The intent of such names is to improve the legibility of mathematical expressions.
049      
050     <P>Example : 
051     <PRE> if ( amount.lt(hundred) ) {
052         cost = amount.times(price); 
053     }</PRE>
054     
055     <P>Several methods in this class are overloaded to perform the same operation with various types:
056    <ul>
057     <li>Decimal
058     <li>long (which will also accept an int)
059     <li>double  (which will also accept a float)
060    </ul>
061      Usually, you should prefer the Decimal form. The long and double forms are convenience methods, which simply call the Decimal 
062      version as part of their implementations; they're intended for cases when you simply wish to specify a hard-coded constant value. 
063     
064     <P>This class extends {@link Number}. This allows other parts of the JDK to treat a <tt>Decimal</tt> just like any other 
065     <tt>Number</tt>.
066    */
067    public final class Decimal extends Number implements Comparable<Decimal>, Serializable {
068    
069      /**
070       The default rounding mode used by this class. 
071       This rounding style results in the least bias.  
072      */ 
073      public static final RoundingMode ROUNDING = RoundingMode.HALF_EVEN;

112      public String toString(){
113        return fAmount.toPlainString();

116      /**
117       Equals, sensitive to scale of the underlying BigDecimal. 
118      
119       <P>That is, <tt>10</tt> and <tt>10.00</tt> are <em>not</em> 
120       considered equal by this method. <b>Such behavior is often undesired; in most 
121       practical cases, it's likely best to use the {@link #eq(Decimal)} method instead.</b>, 
122       which has no such monkey business.
123      
124       <P>This implementation imitates {@link BigDecimal#equals(java.lang.Object)}, 
125       which is also sensitive to the number of decimals (or 'scale'). 
126      */
127      public boolean equals(Object aThat){
128        if (this == aThat) return true;
129        if (! (aThat instanceof Decimal) ) return false;
130        Decimal that = (Decimal)aThat;
131        //the object fields are never null :
132        boolean result = (this.fAmount.equals(that.fAmount) );
133        return result;
134      }
135     
136      public int hashCode(){
137        if ( fHashCode == 0 ) {
138          fHashCode = HASH_SEED;
139          fHashCode = HASH_FACTOR * fHashCode + fAmount.hashCode(); 
140        }
141        return fHashCode;
142      }

144      /**
145       Implements the {@link Comparable} interface. 
146      
147       <P>It's possible to use this method as a general replacement for a large number of methods which compare numbers: 
148       lt, eq, lteq, and so on. However, it's recommended that you use those other methods, since they have greater clarity 
149       and concision.
150      */
151      public int compareTo(Decimal aThat) {
152        final int EQUAL = 0;
153        if ( this == aThat ) return EQUAL;
154        //the object field is never null 
155        int comparison = this.fAmount.compareTo(aThat.fAmount);
156        if ( comparison != EQUAL ) return comparison;
157        return EQUAL;
158      }

164      private static final BigDecimal ZERO_BD = BigDecimal.ZERO;
165      private static final BigDecimal ONE_BD = BigDecimal.ONE;
166      private static final BigDecimal MINUS_ONE_BD = new BigDecimal("-1");

171       <P>Like {@link BigDecimal#ZERO}, this item has no explicit decimal.

174      */

213      /** 
214       Equals (insensitive to number of decimals).
215       That is, <tt>10</tt> and <tt>10.00</tt> are considered equal by this method.
216       
217       <P>Return <tt>true</tt> only if the amounts are equal.
218       This method is <em>not</em> synonymous with the <tt>equals</tt> method, 
219       since the {@link #equals(Object)} method is sensitive to the exact number of decimal places (or, more 
220       precisely, the scale of the underlying BigDecimal.)
221      */

223        return compareAmount(aThat) == 0;
224      }

318      /**
319       Sum a collection of <tt>Decimal</tt> objects.
320       
321       @param aDecimals collection of <tt>Decimal</tt> objects.

323      */
324      public static Decimal sum(Collection<Decimal> aDecimals){
325        Decimal sum = new Decimal(ZERO_BD);
326        for(Decimal decimal : aDecimals){
327          sum = sum.plus(decimal);
328        }

344      /**  
345       Divide this <tt>Decimal</tt> by a divisor.
346       <p>If the division results in a number which will never terminate, then this method 
347       will round the result to 20 decimal places, using the default {@link #ROUNDING}.  
348      */
349      public Decimal div(Decimal aDivisor){
350        BigDecimal newAmount = null;
351        try {
352          newAmount = fAmount.divide(aDivisor.fAmount);
353        }
354        catch(ArithmeticException  ex){
355          // non-terminating decimal
356          // need to apply a policy for where and how to round
357          newAmount = fAmount.divide(aDivisor.fAmount, DECIMALS, ROUNDING);
358        }
359        return new Decimal(newAmount);
360      }
361      public Decimal div(long aDivisor){
362        return div(Decimal.from(aDivisor));
363      }
364      public Decimal div(double aDivisor){  
365        return div(Decimal.from(aDivisor));
366      }

368      /** Return the absolute value of the amount. */
369      public Decimal abs(){
370        return isPlus() ? this : times(-1);
371      }

374      public Decimal negate(){ 
375        return times(-1);

378      /**  Round to an integer value, using the default {@link #ROUNDING} style.  */
379      public Decimal round(){
380        BigDecimal amount = fAmount.setScale(0, ROUNDING);
381        return new Decimal(amount);
382      }

384      /** 
385       Round to 0 or more decimal places, using the default {@link #ROUNDING} style.
386       @param aNumberOfDecimals must 0 or more.  
387      */
388      public Decimal round(int aNumberOfDecimals){
389        if( aNumberOfDecimals < 0 ){
390          throw new IllegalArgumentException("Number of decimals is negative: " + quote(aNumberOfDecimals));
391        }
392        BigDecimal amount = fAmount.setScale(aNumberOfDecimals, ROUNDING);
393        return new Decimal(amount);
394      }
395      
396      /**  
397       Round to 0 or more decimal places, using the given rounding style. 
398       @param aNumberOfDecimals must 0 or more.  
399      */
400      public Decimal round(int aNumberOfDecimals, RoundingMode aRoundingMode){
401        if( aNumberOfDecimals < 0 ){
402          throw new IllegalArgumentException("Number of decimals is negative: " + quote(aNumberOfDecimals));
403        }
404        BigDecimal amount = fAmount.setScale(aNumberOfDecimals, aRoundingMode);
405        return new Decimal(amount);
406      }

408      /**
409       Round a number to the nearest multiple of the given interval.
410       For example:
411       <tt>
412         Decimal amount = Decimal.from("1710.12");
413         amount.round2(0.05); // 1710.10
414         amount.round2(100);  // 1700
415       </tt>
416       @param aInterval must be greater than zero
417      */
418      public Decimal round2(Decimal aInterval){
419        if( ! aInterval.isPlus() ){
420          throw new IllegalArgumentException("Interval is negative or zero : " + quote(aInterval));
421        }
422        BigDecimal result = fAmount.divide(aInterval.fAmount).setScale(0, ROUNDING).multiply(aInterval.fAmount);
423        return new Decimal(result);
424      }
425      public Decimal round2(long aInterval){
426        return round2(Decimal.from(aInterval));    
427      }
428      public Decimal round2(double aInterval){
429        return round2(Decimal.from(aInterval));    
430      }

432      /**
433       Raise this number to an <b>integral</b> power; the power can be of either sign.
434       
435       <P>Special cases regarding 0:
436       <ul>
437         <li> <tt>0^-n</tt> is undefined (<tt>n > 0</tt>).
438         <li> <tt>x^0<tt/> always returns 1, even for <tt>x = 0</tt>.
439       </ul>
440       
441       @param aPower is in the range -999,999,999..999,999,999, inclusive. (This reflects a restriction on 
442       the underlying BigDecimal.)
443      */
444      public Decimal pow(int aPower){
445        BigDecimal newAmount = null;
446        if (aPower == 0){
447          newAmount = ONE_BD;
448        }
449        else if (aPower == 1){
450          newAmount = fAmount;
451        }
452        else if (aPower > 0){
453          newAmount = fAmount.pow(aPower);
454        }
455        else if (aPower < 0){
456          newAmount = fAmount.pow(-1 * aPower);
457          newAmount = ONE_BD.divide(newAmount);
458        }
459        return new Decimal(newAmount);  
460      }

522      /**
523       Always treat de-serialization as a full-blown constructor, by
524       validating the final state of the de-serialized object.
525      */  
526      private void readObject(
527        ObjectInputStream aInputStream
528      ) throws ClassNotFoundException, IOException {
529        //always perform the default de-serialization first
530        aInputStream.defaultReadObject();
531        //defensive copy for mutable date field
532        //BigDecimal is not technically immutable, since its non-final
533        fAmount = new BigDecimal( fAmount.toPlainString() );
534        //ensure that object state has not been corrupted or tampered with maliciously
535        validateState();
536      }

539        //perform the default serialization for all non-transient, non-static fields
540        aOutputStream.defaultWriteObject();
541      }

543      private void validateState(){
544        if( fAmount == null ) {
545          throw new IllegalArgumentException("Amount cannot be null");
546        }
547      }

549      /** Ignores scale: 0 same as 0.00 */
550      private int compareAmount(Decimal aThat){
551        return this.fAmount.compareTo(aThat.fAmount);
552      }
